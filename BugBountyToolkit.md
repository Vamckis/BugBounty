- docker pull hackersploit/bugbountytoolkit
- docker run -it hackersploit/bugbountytoolkit /bin/bash

- git clone https://github.com/AlexisAhmed/BugBountyToolkit.git
- cd BugBountyToolkit
- chmod +x install.sh
- ./install.sh
